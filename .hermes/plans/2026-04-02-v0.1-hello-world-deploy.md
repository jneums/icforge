# ICForge v0.1 — "Hello World Deploy" Implementation Plan

> **For Hermes:** Use subagent-driven-development skill to implement this plan task-by-task.

**Goal:** Deploy a static HTML/JS site to IC mainnet via `icforge deploy` and return a `<canister-id>.ic0.app` URL.

**Architecture:** CLI authenticates via GitHub OAuth → backend generates a custodial IC identity per user → CLI uploads build artifacts → backend calls IC management canister to create canister + install code + sync assets → returns live URL. Backend is Rust/Axum on Render. Dashboard is React+Vite placeholder (scaffold only for v0.1).

**Tech Stack:** Rust (Axum, ic-agent 0.47, sqlx/SQLite), TypeScript (Commander CLI, React+Vite dashboard), GitHub OAuth

```rust
        .connect(database_url)
        .await?;

    Ok(pool)
```

```rust
        let (status, message) = match self {
            AppError::BadRequest(msg) => (StatusCode::BAD_REQUEST, msg),
            AppError::Unauthorized(msg) => (StatusCode::UNAUTHORIZED, msg),

            }
        };
        (status, Json(json!({ "error": message }))).into_response()

}

impl From<sqlx::Error> for AppError {
```

```rust
// backend/src/auth.rs
use jsonwebtoken::{encode, decode, Header, Validation, EncodingKey, DecodingKey};
use serde::{Deserialize, Serialize};
use chrono::Utc;

#[derive(Debug, Serialize, Deserialize)]
pub struct Claims {
    pub sub: String,      // user ID
    pub exp: usize,       // expiry timestamp
    pub iat: usize,       // issued at
}

pub fn create_token(user_id: &str, secret: &str) -> Result<String, jsonwebtoken::errors::Error> {
    let now = Utc::now().timestamp() as usize;
    let claims = Claims {
        sub: user_id.to_string(),
        exp: now + 3600 * 24 * 7, // 7 days
        iat: now,
    };
    encode(&Header::default(), &claims, &EncodingKey::from_secret(secret.as_bytes()))
}

pub fn verify_token(token: &str, secret: &str) -> Result<Claims, jsonwebtoken::errors::Error> {
    let data = decode::<Claims>(token, &DecodingKey::from_secret(secret.as_bytes()), &Validation::default())?;
    Ok(data.claims)
}

#[derive(Deserialize)]
pub struct GitHubTokenResponse {
    pub access_token: String,
}

#[derive(Deserialize)]
pub struct GitHubUser {
    pub id: i64,
    pub login: String,
    pub email: Option<String>,
    pub name: Option<String>,
    pub avatar_url: Option<String>,
}

pub async fn exchange_github_code(
    client_id: &str,
    client_secret: &str,
    code: &str,
) -> Result<String, reqwest::Error> {
    let client = reqwest::Client::new();
    let resp: GitHubTokenResponse = client
        .post("https://github.com/login/oauth/access_token")
        .header("Accept", "application/json")
        .json(&serde_json::json!({
            "client_id": client_id,
            "client_secret": client_secret,
            "code": code,
        }))
        .send()
        .await?
        .json()
        .await?;
    Ok(resp.access_token)
}

pub async fn get_github_user(access_token: &str) -> Result<GitHubUser, reqwest::Error> {
    let client = reqwest::Client::new();
    client
        .get("https://api.github.com/user")
        .header("Authorization", format!("Bearer {access_token}"))
        .header("User-Agent", "ICForge")
        .send()
        .await?
        .json()
        .await
}
```

```rust
// backend/src/identity.rs
use ic_agent::identity::{BasicIdentity, Identity};
use ring::signature::Ed25519KeyPair;
use ring::rand::SystemRandom;

/// Generate a new Ed25519 identity and return (pem_string, principal_text)
pub fn generate_identity() -> Result<(String, String), Box<dyn std::error::Error>> {
    let rng = SystemRandom::new();
    let pkcs8_doc = Ed25519KeyPair::generate_pkcs8(&rng)
        .map_err(|e| format!("Key generation failed: {e}"))?;

    // Encode as PEM
    let pem = pem::encode(&pem::Pem::new("PRIVATE KEY", pkcs8_doc.as_ref().to_vec()));

    // Derive principal
    let identity = BasicIdentity::from_pem(pem.as_bytes())
        .map_err(|e| format!("Failed to create identity from PEM: {e}"))?;
    let principal = identity.sender()
        .map_err(|e| format!("Failed to get principal: {e}"))?;

    Ok((pem, principal.to_text()))
}

/// Load an identity from stored PEM
pub fn load_identity(pem_str: &str) -> Result<BasicIdentity, Box<dyn std::error::Error>> {
    Ok(BasicIdentity::from_pem(pem_str.as_bytes())?)
}
```

```typescript
// cli/src/api.ts
import { getToken } from "./auth.js";

const API_URL = process.env.ICFORGE_API_URL ?? "https://icforge-backend.onrender.com";

export function getApiUrl(): string {
    return API_URL;
}

export async function apiFetch(path: string, options: RequestInit = {}): Promise<Response> {
    const token = getToken();
    const headers: Record<string, string> = {
        "Content-Type": "application/json",
        ...((options.headers as Record<string, string>) ?? {}),
    };
    if (token) {
        headers["Authorization"] = `Bearer ${token}`;
    }
    return fetch(`${API_URL}${path}`, { ...options, headers });
}
```

**Step 2: Implement login command**

```typescript
// cli/src/commands/login.ts
import { createServer } from "http";
import open from "open";
import chalk from "chalk";
import ora from "ora";
import { saveToken } from "../auth.js";
import { getApiUrl } from "../api.js";

export async function loginCommand() {
    const spinner = ora("Waiting for browser authentication...").start();

    // Start local HTTP server to receive callback
    const { token, cleanup } = await new Promise<{ token: string; cleanup: () => void }>((resolve, reject) => {
        const server = createServer(async (req, res) => {
            const url = new URL(req.url!, `http://localhost`);
            if (url.pathname === "/callback") {
                const jwt = url.searchParams.get("token");
                if (jwt) {
                    res.writeHead(200, { "Content-Type": "text/html" });
                    res.end("<html><body><h1>✅ Logged in to ICForge!</h1><p>You can close this tab.</p></body></html>");
                    resolve({ token: jwt, cleanup: () => server.close() });
                } else {
                    res.writeHead(400);
                    res.end("Missing token");
                    reject(new Error("No token received"));
                }
            }
        });

        server.listen(0, () => {
            const port = (server.address() as any).port;
            const authUrl = `${getApiUrl()}/api/v1/auth/login?redirect=http://localhost:${port}/callback`;
            open(authUrl);
        });

        // Timeout after 2 minutes
        setTimeout(() => {
            server.close();
            reject(new Error("Login timed out"));
        }, 120_000);
    });

    cleanup();

    // Save token
    saveToken(token, "", 7 * 24 * 3600); // 7 day expiry
    spinner.succeed(chalk.green("Logged in to ICForge!"));
}
```

**Step 1: Create ic_client.rs**

```rust
// backend/src/ic_client.rs
use ic_agent::{Agent, agent::http_transport::reqwest_transport::ReqwestTransport, identity::BasicIdentity};
use candid::{Principal, Encode, CandidType};
use serde::Deserialize;

const IC_URL: &str = "https://ic0.app";

pub struct IcClient {
    agent: Agent,
}

#[derive(CandidType)]
struct CreateCanisterArgs {
    settings: Option<CanisterSettings>,
}

#[derive(CandidType)]
struct CanisterSettings {
    controllers: Option<Vec<Principal>>,
    // Subnet selection would go here in the future
}

#[derive(CandidType)]
struct InstallCodeArgs {
    mode: InstallMode,
    canister_id: Principal,
    wasm_module: Vec<u8>,
    arg: Vec<u8>,
}

#[derive(CandidType)]
enum InstallMode {
    #[serde(rename = "install")]
    Install,
    #[serde(rename = "upgrade")]
    Upgrade,
    #[serde(rename = "reinstall")]
    Reinstall,
}

impl IcClient {
    pub async fn new(identity_pem: &str) -> Result<Self, Box<dyn std::error::Error>> {
        let identity = BasicIdentity::from_pem(identity_pem.as_bytes())?;
        let transport = ReqwestTransport::create(IC_URL)?;
        let agent = Agent::builder()
            .with_transport(transport)
            .with_identity(identity)
            .build()?;
        // Fetch root key only on local replica, not mainnet
        // agent.fetch_root_key().await?;
        Ok(Self { agent })
    }

    pub async fn create_canister(&self) -> Result<Principal, Box<dyn std::error::Error>> {
        // Call management canister to create a new canister
        let management = Principal::management_canister();
        let args = CreateCanisterArgs { settings: None };
        let response = self.agent
            .update(&management, "create_canister")
            .with_arg(Encode!(&args)?)
            .call_and_wait()
            .await?;
        
        #[derive(CandidType, Deserialize)]
        struct CreateResult {
            canister_id: Principal,
        }
        let result = candid::Decode!(&response, CreateResult)?;
        Ok(result.canister_id)
    }

    pub async fn install_code(
        &self,
        canister_id: Principal,
        wasm: Vec<u8>,
        upgrade: bool,
    ) -> Result<(), Box<dyn std::error::Error>> {
        let management = Principal::management_canister();
        let mode = if upgrade { InstallMode::Upgrade } else { InstallMode::Install };
        let args = InstallCodeArgs {
            mode,
            canister_id,
            wasm_module: wasm,
            arg: Encode!(&())?,
        };
        self.agent
            .update(&management, "install_code")
            .with_arg(Encode!(&args)?)
            .call_and_wait()
            .await?;
        Ok(())
    }

    pub async fn canister_status(&self, canister_id: Principal) -> Result<String, Box<dyn std::error::Error>> {
        // Simplified — just check if canister is running
        #[derive(CandidType)]
        struct StatusArgs {
            canister_id: Principal,
        }
        let management = Principal::management_canister();
        let _response = self.agent
            .update(&management, "canister_status")
            .with_arg(Encode!(&StatusArgs { canister_id })?)
            .call_and_wait()
            .await?;
```

**Step 2: Implement deploy pipeline in deploy.rs**

```rust
// High-level deploy flow:
pub async fn execute_deploy(
    db: &DbPool,
    deployment_id: &str,
    user_pem: &str,
    canister_record: &CanisterRecord,
    wasm_bytes: Vec<u8>,
) -> Result<String, AppError> {
    let ic = IcClient::new(user_pem).await
        .map_err(|e| AppError::Internal(format!("IC client init failed: {e}")))?;

    log_deploy(db, deployment_id, "info", "Starting deployment...").await?;

    // 1. Create canister if needed
    let canister_id = if let Some(cid) = &canister_record.canister_id {
        log_deploy(db, deployment_id, "info", &format!("Upgrading existing canister {cid}")).await?;
        Principal::from_text(cid).map_err(|e| AppError::Internal(e.to_string()))?
    } else {
        log_deploy(db, deployment_id, "info", "Creating new canister...").await?;
        let cid = ic.create_canister().await
            .map_err(|e| AppError::Internal(format!("Failed to create canister: {e}")))?;
        log_deploy(db, deployment_id, "info", &format!("Created canister: {cid}")).await?;
        // Save canister ID to DB
        update_canister_id(db, &canister_record.id, &cid.to_text()).await?;
        cid
    };

    // 2. Install code
    let is_upgrade = canister_record.canister_id.is_some();
    log_deploy(db, deployment_id, "info", &format!(
        "{} code ({} bytes)...",
        if is_upgrade { "Upgrading" } else { "Installing" },
        wasm_bytes.len()
    )).await?;

    match ic.install_code(canister_id, wasm_bytes, is_upgrade).await {
        Ok(()) => {
            log_deploy(db, deployment_id, "info", "Code installed successfully").await?;
        }
        Err(e) => {
            let msg = format!("install_code failed: {e}");
            log_deploy(db, deployment_id, "error", &msg).await?;
            update_deployment_status(db, deployment_id, "failed", Some(&msg)).await?;
            return Err(AppError::Internal(msg));
        }
    }

    // 3. Verify
    log_deploy(db, deployment_id, "info", "Verifying deployment...").await?;
    let url = format!("https://{}.ic0.app", canister_id);
    log_deploy(db, deployment_id, "info", &format!("🚀 Live at: {url}")).await?;

    update_deployment_status(db, deployment_id, "live", None).await?;
    Ok(url)
}
```
